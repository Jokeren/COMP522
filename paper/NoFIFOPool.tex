\begin{algo}[!ht]
\caption{Non-FIFO implementation of SCPool: Functions.} 
\label{alg:non-fifo}
\scriptsize
\begin{minipage}[t]{0.48\textwidth}
\begin{distribalgo}[1]
\setcounter{ALC@line}{\value{alg:non-fifo:lines}}
\smallskip
\INDENT {{\bf Function produce}(Task t):}
	\STATE chunk $\leftarrow$ producerCtx.chunk
	\INDENT {{\bf if} (chunk $= \bot$) {\bf then}}
		\STATE chunk $\leftarrow$ chunkPool.getChunk()
		\STATE {\bf if} (chunk $= \bot$) {\bf then return FULL} \comment {no available chunks}
		\STATE chunk.owner $\leftarrow$ consumerId
		\STATE node $\leftarrow$ new node with node.idx $=0$ and node.c $=$ chunk
		
		%\STATE node $\leftarrow$ new node; node.idx $\leftarrow 0$; node.chunk $\leftarrow$ chunk
		\STATE chunkLists[producerCtx.producerId].{\bf append}(node)
		%\STATE {\bf FetchAndInc}(chunkCounters[producerCtx.producerId])
		\STATE producerCtx.chunk $\leftarrow$ chunk; producerCtx.prodIdx $\leftarrow 0$ 
	\ENDINDENT
	\STATE chunk.tasks[producerCtx.prodIdx] $\leftarrow$ t
	\STATE producerCtx.prodIdx++
	\INDENT {{\bf if}(producerCtx.prodIdx $=$ CHUNK\_SIZE) {\bf then}}
	  \STATE producerCtx.chunk $\leftarrow \bot$ \comment {the chunk is full}
	\ENDINDENT
	\STATE {\bf return SUCCESS}
\ENDINDENT

\medskip

\INDENT {{\bf Function takeTask}(Node n):}
  \STATE chunk $\leftarrow$ n.c
  \STATE {{\bf if} (chunk $= \bot$) {\bf then return $\bot$} \comment{this chunk has been stolen}}
 
  \STATE task $\leftarrow$ chunk.tasks[n.idx + 1]
  \STATE {\bf if} (task $= \bot$) {\bf then return} $\bot$ \comment{no inserted tasks}
 	
 	\smallskip 
  \STATE n.idx++ \comment {tell the world you're going to take a task from idx}
  \INDENT {{\bf if} (chunk.owner $=$ consumerId) {\bf then} \comment {common case}}
 		\STATE chunk.tasks[n.idx] $\leftarrow \bot_2$
  	\INDENT {{\bf if}(n.idx $+1 = $ CHUNK\_SIZE) {\bf then} \comment {finished the chunk}}
  		\STATE {\bf reclaimChunk}(chunk); n.c $\leftarrow \bot$
  		\STATE currentNode $\leftarrow \bot$
  	\ENDINDENT
  	\STATE {\bf return} task 
  \ENDINDENT
  
  \smallskip
  \comment {the chunk has been stolen, CAS the last task and go away}
 	\STATE success $\leftarrow$ (task $\neq \bot_2 \wedge $ CAS(chunk.tasks[n.idx], task, $\bot_2$))
	\STATE n.c $\leftarrow \bot$; currentNode $\leftarrow \bot$
 	\STATE {\bf return} (success) ? task : $\bot$
\ENDINDENT

\setcounter{alg:non-fifo:lines}{\value{ALC@line}} % store the line number
\end{distribalgo}

\begin{distribalgo}[1]
\setcounter{ALC@line}{\value{alg:non-fifo:lines}}
\smallskip

\INDENT {{\bf Function consume}():}
  \INDENT {{\bf if}(currentNode $\neq \bot$) {\bf then} \comment {common case}}
		\STATE t $\leftarrow$ {\bf takeTask}(currentNode)
		\STATE {\bf if} (t $\neq \bot$) {\bf then return} t
	\ENDINDENT
	\comment {Iterate over other chunk lists}
	\INDENT {{\bf foreach} cl {\bf in} chunkLists {\bf do}}
	%\INDENT {{\bf for} i $= 0, \ldots, $ NUM\_PRODUCERS {\bf do}}
		%\STATE currentEntry $\leftarrow$ (currentEntry+1) \% NUM\_PRODUCERS
  	\INDENT {{\bf foreach} node {\bf in} cl {\bf do}}
  		%\STATE currentNode $\leftarrow$ chunkList.{\bf head}()
  		%\INDENT {{\bf if} (node $\neq \bot$) {\bf then}}
  		\STATE t $\leftarrow$ {\bf takeTask}(node)
			\STATE {\bf if} (t $\neq \bot$) {\bf then} currentNode $\leftarrow$ node; {\bf return} t
  		%\ENDINDENT
  	\ENDINDENT
	\ENDINDENT
	\STATE currentNode $\leftarrow \bot$; {\bf return} $\bot$
\ENDINDENT

\medskip

\INDENT {{\bf Function steal}(SCPool from):}
	\STATE nodeToSteal $\leftarrow$ a non-empty node from the longest list
	\STATE {\bf return} {\bf stealChunk}(nodeToSteal, from.consumerId)
%		
%	
%	\STATE idx $\leftarrow$ chooseEntryForSteal(from)
%	\STATE {\bf if}(from.chunkCounters[idx] $= 0$) {\bf then return} $\bot$
%	\STATE {\bf return} {\bf stealChunk}(from.prodEntries[idx], idx, consumerId)
\ENDINDENT

\medskip

\INDENT {{\bf Function stealChunk}(Node prevNode, int prevOwner):}
  \STATE c $\leftarrow$ prevNode.c;
  \STATE {\bf if} ({\bf isEmpty}(prevNode)) {\bf then return} $\bot$

  \STATE chunkLists[stealEntry].{\bf append}(prevNode)
  \INDENT {{\bf if} ({\bf CAS}(c.owner, prevOwner, consumerId) $=$ false) {\bf then}}
 		\STATE chunkLists[stealEntry].{\bf remove}(prevNode)
 		\STATE {\bf return} $\bot$ \comment {failed to steal}
	\ENDINDENT
	
	\smallskip
	\STATE newNode $\leftarrow$ copy of prevNode
	\STATE newNode.c.owner $\leftarrow$ consumerId
	%\STATE newNode $\leftarrow$ new node; newNode.chunk $\leftarrow$ c
	%\STATE newNode.idx $\leftarrow$ from.idx; newNode.owner $\leftarrow$ consumerIdx
	\STATE replace prevNode with newNode in chunkLists[stealEntry]
	\STATE prevNode.c $\leftarrow \bot$; 
	
	\smallskip
	\comment {try to take the task at idx+1 using CAS}
	\STATE task $\leftarrow$ c.tasks[idx+1]
	\STATE {\bf if} (task $= \bot$) {\bf then return} $\bot$ \comment {still no task at idx+1}
	\INDENT {{\bf if} (task $= \bot_2 \vee $ !CAS(c.tasks[idx+1], task, $\bot_2$)) {\bf then}} 
		\STATE task $\leftarrow \bot$
	\ENDINDENT
	\STATE newNode.idx $\leftarrow$ newNode.idx+1
	\STATE {\bf return} task
\ENDINDENT

\setcounter{alg:non-fifo:lines}{\value{ALC@line}}
\end{distribalgo}
\end{minipage}
\end{algo}
